MODELS.PY

from django.db import models

class Machine(models.Model):
    name = models.CharField(max_length=100)

    def __str__(self):
        return self.name

class MachineElement(models.Model):
    machine = models.ForeignKey(Machine, on_delete=models.CASCADE, related_name='elements')
    sensor_name = models.CharField(max_length=100)
    tag = models.CharField()
    server_ip = models.CharField(max_length=100) 

    element_type = models.CharField(max_length=20, blank=True)
    upper_limit = models.FloatField(null=True, blank=True)
    lower_limit = models.FloatField(max_length=50, blank=True)
    organization_id = models.IntegerField(null=True, blank=True)
    active = models.BooleanField(default=True)

    def __str__(self):
        return f"{self.element_name} ({self.machine.name})"

from django.shortcuts import render
from .models import Machine

def machine_list(request):
    machines = Machine.objects.prefetch_related('elements').all()
    return render(request, 'machine_monitor/machine_list.html', {'machines': machines})
